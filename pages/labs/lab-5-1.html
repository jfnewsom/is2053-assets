class City:

    # --- Constructor ---   [CP1]
    __init__(self, name, city_type, tagline)
        self.__name          = name
        self.__city_type     = city_type
        self.__tagline       = tagline
        self.__connections   = {}     (empty dict to start)

    # --- Getter Methods ---   [CP2]
    get_name(self)                    returns self.__name
    get_type(self)                    returns self.__city_type
    get_tagline(self)                 returns self.__tagline

    # --- Connection Methods ---   [CP3]
    add_connection(self, city, dist)  stores in self.__connections
    get_distance_to(self, city)       returns distance or -1
    get_destination_names(self)       returns list of connected names

    # --- Status Methods ---   [CP4]
    is_win_city(self)                 True if type == 'WIN'
    is_dead_end(self)                 True if type == 'DEAD_END' or 'WRONG_WAY'
    get_arrival_message(self)         returns message string based on city type
    display_destinations(self)        prints numbered menu of connections
    __str__(self)                     returns 'San Antonio (START)'

Step-by-Step Instructions

def add_connection(self, city_name, distance):
    """Add a connection to another city."""
    self.__connections[city_name] = distance
  1. Define get_distance_to(self, city_name) that returns the distance, or -1 if not connected:
def get_distance_to(self, city_name):
    """Get distance to a connected city. Returns -1 if not connected."""
    return self.__connections.get(city_name, -1)
  1. Define get_destination_names(self) that returns a list of connected city names:
def get_destination_names(self):
    """Return list of connected city names."""
    return list(self.__connections.keys())

def display_destinations(self):
    """Display available destinations from this city."""
    print(f'From {self.__name}, you can travel to:')
    option_num = 1
    for dest_name, distance in self.__connections.items():
        print(f'  {option_num}. {dest_name} - {distance} miles')
        option_num = option_num + 1
  1. Define __str__(self) that returns a string like 'San Antonio (START)':
def __str__(self):
    """Return string representation of the city."""
    return f'{self.__name} ({self.__city_type})'
